# CICESE-BI Environment Configuration Template
# Copy this content to a file named .env in your project root
# DO NOT commit .env to version control!

# Database Configuration
# For cPanel: Use the full database name (includes username prefix)

DB_NAME=your_cpanel_username_cicese
DB_USER=your_cpanel_username_ciceseuser

# Site Configuration
